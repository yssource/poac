11calc(StringRef a, StringRef b) {
12 const usize asize = a.size();
13 const usize bsize = b.size();
14
15 // for all i and j, d[i,j] will hold the Levenshtein distance between the
16 // first i characters of s and the first j characters of t
17 Vec<Vec<usize>> d(asize + 1, Vec<usize>(bsize + 1));
18 d[0][0] = 0;
19
20 // source prefixes can be transformed into empty string by dropping all
21 // characters
22 for (usize i = 1; i <= asize; ++i) {
23 d[i][0] = i;
24 }
25
26 // target prefixes can be reached from empty source prefix by inserting every
27 // character
28 for (usize j = 1; j <= bsize; ++j) {
29 d[0][j] = j;
30 }
31
32 for (usize i = 1; i <= asize; ++i) {
33 for (usize j = 1; j <= bsize; ++j) {
34 const usize substCost = a[i - 1] == b[j - 1] ? 0 : 1;
35 d[i][j] = std::min({
36 d[i - 1][j] + 1, // deletion
37 d[i][j - 1] + 1, // insertion
38 d[i - 1][j - 1] + substCost // substitution
39 });
40 }
41 }
42
43 return d[asize][bsize];
44}

55Option<StringRef>
56find_similar_str(StringRef lhs, std::span<const StringRef> candidates) {
57 // We need to check if `Candidates` has the exact case-insensitive string
58 // because the Levenshtein distance match does not care about it.
59 for (StringRef c : candidates) {
60 if (equals_insensitive(lhs, c)) {
61 return c;
62 }
63 }
64
65 // Keep going with the Levenshtein distance match.
66 // If the LHS size is less than 3, use the LHS size minus 1 and if not,
67 // use the LHS size divided by 3.
68 usize length = lhs.size();
69 usize max_dist = length < 3 ? length - 1 : length / 3;
70
71 Option<std::pair<StringRef, usize>> similar_str = None;
72 for (StringRef c : candidates) {
73 usize cur_dist = calc(lhs, c);
74 if (cur_dist <= max_dist) {
75 if (!similar_str.has_value()) {
76 // The first similar string found.
77 similar_str = {c, cur_dist};
78 } else if (cur_dist < similar_str->second) {
79 // More similar string found.
80 similar_str = {c, cur_dist};
81 }
82 }
83 }
84
85 if (similar_str.has_value()) {
86 return similar_str->first;
87 } else {
88 return None;
89 }
90}

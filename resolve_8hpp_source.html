99template <typename SinglePassRange>
100bool
101duplicate_loose(const SinglePassRange& rng) {
102 const auto first = std::begin(rng);
103 const auto last = std::end(rng);
104 return std::find_if(

106 [&](const auto& x) {
107 return std::count_if(

109 [&](const auto& y) {

112 )

114 }

116 != last;

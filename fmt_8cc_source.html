18[[nodiscard]] Result<void>
19fmt(StringRef args) {
20 Vec<Path> targets{};
21 for (StringRef d : directories) {
22 if (!fs::exists(d)) {
23 spdlog::trace("Directory `{}` not found; skipping ...", d);
24 continue;
25 }
26
27 for (StringRef e : extensions) {
28 for (StringRef p : patterns) {
29 const String search = format(::fmt::runtime(p), d, e);
30 const Vec<Path> search_glob = glob::rglob(search);
31 if (search_glob.empty()) {
32 spdlog::trace("Glob `{}` not found; skipping ...", search);
33 continue;
34 }
35 spdlog::trace("Glob `{}` found!", search);
36 append(targets, search_glob);
37 }
38 }
39 }
40 if (targets.empty()) {
41 spdlog::info("no targets found.");
42 return Ok();
43 }
44
45 const String clang_format =
46 format("clang-format {} {}", args, ::fmt::join(targets, " "));
47 spdlog::trace("Executing `{}`", clang_format);
48 if (const i32 code = util::shell::Cmd(clang_format).exec_no_capture();
49 code != 0) {
50 spdlog::info("");
51 return Err<SubprocessFailed>("clang-format", code);
52 }
53 return Ok();
54}

56[[nodiscard]] Result<void>
57exec(const Options& opts) {
58 spdlog::trace("Checking if `clang-format` command exists ...");
59 if (!util::shell::has_command("clang-format")) {
60 return Err<ClangFormatNotFound>();
61 }
62
63 spdlog::trace("Checking if required config exists ...");
64 Try(core::validator::required_config_exists().map_err(to_anyhow));
65
66 spdlog::trace("Parsing the manifest file ...");
67 // TODO(ken-matsui): parse as a static type rather than toml::value
68 const toml::value manifest = toml::parse(data::manifest::name);
69 String name = toml::find<String>(manifest, "package", "name");
70 if (name.empty()) {
71 log::warn("project name is empty; try setting anything you want.");
72 name = "<empty>";
73 }
74
75 String args = "--style=file --fallback-style=LLVM -Werror ";
76 if (util::verbosity::is_verbose()) {
77 args += "--verbose ";
78 }
79 if (opts.check.value()) {
80 args += "--dry-run";
81 } else {
82 args += "-i";
83 log::status("Formatting", name);
84 }
85 return fmt(args);
86}

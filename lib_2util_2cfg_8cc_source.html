75std::pair<Lexer::size_type, Option<Token>>
76Lexer::tokenize(size_type index_) const {
77 if (index_ >= this->str.size()) {
78 return generate_token(index_, None);
79 }
80 const char one = this->one(index_);
81 switch (one) {
82 case ' ':
83 do {
84 this->step(index_);
85 } while (this->one(index_) == ' ');
86 return tokenize(index_);
87 case '(':
88 [[fallthrough]];
89 case ')':
90 [[fallthrough]];
91 case ',':
92 [[fallthrough]];
93 case '=':
94 this->step(index_);
95 return generate_token(index_, String{one});
96 case '>':
97 [[fallthrough]];
98 case '<':
99 this->step(index_);
100 return analyze_two_phrase(index_, one);
101 case '"':
102 return string(index_);
103 default:
104 return ident(index_);
105 }
106}

121std::pair<Lexer::size_type, Token>
122Lexer::string(size_type index_) const {
123 this->step(index_);
124 const size_type start = index_;
125 while (this->one(index_) != '"') {
126 this->step(index_);
127 if (index_ >= this->str.size()) {
128 String msg;
129 msg += String(start - 1, ' ');
130 msg += "^";
131 msg += String(this->str.size() - start, '-');
132 msg += " unterminated string";
133 throw cfg::string_error(String(this->str) + "\n" + msg);
134 }
135 }
136 StringRef s = this->str.substr(start, index_ - start);
137 this->step(index_);
138 return {this->diff_step(index_), Token{Token::String, s}};
139}

141std::pair<Lexer::size_type, Token>
142Lexer::ident(size_type index_) const {
143 if (!is_ident_start(this->one(index_))) {
144 String msg;
145 msg += String(index_, ' ');
146 msg += "^ unexpected character";
147 throw cfg::ident_error(String(this->str) + "\n" + msg);
148 }
149 const size_type start = index_;
150 this->step(index_);
151 while (is_ident_rest(this->one(index_))) {
152 this->step(index_);
153 }
154
155 StringRef s = this->str.substr(start, index_ - start);
156 if (const auto ident = to_ident(s)) {
157 return {this->diff_step(index_), Token{Token::Ident, ident.value()}};
158 } else {
159 String msg;
160 msg += String(start, ' ');
161 msg += "^";
162 msg += String(index_ - start - 1, '-');
163 msg += " unknown identify";
164 throw cfg::ident_error(String(this->str) + "\n" + msg);
165 }
166}

215bool
216CfgExpr::match() const {
217 switch (this->kind) {
218 case Kind::cfg:
219 return std::get<expr_type>(this->expr)->match();
220 case Kind::not_:
221 return !(std::get<expr_type>(this->expr)->match());
222 case Kind::all: {
223 bool res = true;
224 for (const CfgExpr& c : std::get<expr_list_type>(this->expr)) {
225 res &= c.match();
226 }
227 return res;
228 }
229 case Kind::any: {
230 bool res = false;
231 for (const CfgExpr& c : std::get<expr_list_type>(this->expr)) {
232 res |= c.match();
233 }
234 return res;
235 }
236 case Kind::value:
237 return this->match(std::get<Cfg>(this->expr));
238 default:
239 unreachable();
240 }
241}

341CfgExpr
342Parser::expr() {
343 if (const auto token = lexer.peek(); !token.has_value()) {
344 throw cfg::expression_error("expected start of a cfg expression");
345 } else if (token->kind == Token::Ident) {
346 if (token->get_ident() == Token::ident::all
347 || token->get_ident() == Token::ident::any) {
348 this->lexer.next();
349 this->eat_left_paren(token->get_ident());
350 Vec<CfgExpr> e;
351 do { // `all` and `any` need at least one expression.
352 e.emplace_back(this->expr());
353 if (!this->r_try(Token::Comma)) {
354 this->eat_right_paren();
355 break;
356 }
357 } while (!r_try(Token::RightParen));
358 if (token->get_ident() == Token::ident::all) {
359 return CfgExpr{CfgExpr::all, std::move(e)};
360 } else {
361 return CfgExpr{CfgExpr::any, std::move(e)};
362 }
363 } else if (token->get_ident() == Token::ident::not_ || token->get_ident() == Token::ident::cfg) {
364 this->lexer.next();
365 this->eat_left_paren(token->get_ident());
366 CfgExpr&& e = this->expr();
367 this->eat_right_paren();
368 if (token->get_ident() == Token::ident::not_) {
369 return CfgExpr{CfgExpr::not_, std::make_unique<CfgExpr>(std::move(e))};
370 } else {
371 return CfgExpr{CfgExpr::cfg, std::make_unique<CfgExpr>(std::move(e))};
372 }
373 }
374 }
375 return CfgExpr{CfgExpr::value, this->cfg()};
376}

378Cfg
379Parser::cfg() {
380 const usize index = lexer.index;
381 if (const auto token = lexer.next(); !token.has_value()) {
382 String msg = String(index + 1, ' ');
383 msg += " ^ expected operator, but cfg expression ended";
384 throw cfg::syntax_error(String(lexer.str) + "\n" + msg);
385 } else if (token->kind == Token::Ident) {
386 if (this->r_try(Token::Equals)) {
387 return this->cfg_str(token->get_ident(), Cfg::Op::Equals);
388 } else if (this->r_try(Token::Gt)) {
389 return this->cfg_str(lexer.index, token->get_ident(), Cfg::Op::Gt);
390 } else if (this->r_try(Token::GtEq)) {
391 return this->cfg_str(lexer.index, token->get_ident(), Cfg::Op::GtEq);
392 } else if (this->r_try(Token::Lt)) {
393 return this->cfg_str(lexer.index, token->get_ident(), Cfg::Op::Lt);
394 } else if (this->r_try(Token::LtEq)) {
395 return this->cfg_str(lexer.index, token->get_ident(), Cfg::Op::LtEq);
396 }
397 }
398 String msg = String(lexer.index + 1, ' ');
399 msg += "^ expected operator";
400 throw cfg::syntax_error(String(lexer.str) + "\n" + msg);
401}

458void
459Parser::eat_left_paren(Token::ident prev) {
460 const usize index = lexer.index;
461 if (const auto token = lexer.next()) {
462 if (token->kind != Token::LeftParen) {
463 String msg = String(index, ' ');
464 msg += "^ excepted '(' after `" + to_string(prev) + "`";
465 throw cfg::syntax_error(String(lexer.str) + "\n" + msg);
466 }
467 } else {
468 String msg = String(index, ' ');
469 msg += "^ expected '(', but cfg expression ended";
470 throw cfg::syntax_error(String(lexer.str) + "\n" + msg);
471 }

484 } else {

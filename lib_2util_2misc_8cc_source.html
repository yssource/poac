46expand_user() {
47 Option<String> home = dupenv("HOME");
48 if (home || (home = dupenv("USERPROFILE"))) {
49 return Ok(home.value());
50 } else {
51 const auto home_drive = dupenv("HOMEDRIVE");
52 const auto home_path = dupenv("HOMEPATH");
53 if (home_drive && home_path) {
54 return Ok(home_drive.value() + home_path.value());
55 }
56 return Err("could not get home directory");
57 }

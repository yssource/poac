35[[nodiscard]] Result<std::pair<Path, String>>
36fetch_impl(const resolve::Package& package) noexcept {
37 try {
38 const auto [download_link, sha256sum] =
39 Try(get_download_link(package).map_err(to_anyhow));
40 log::debug("downloading from `{}`", download_link);
41 const Path archive_path = get_archive_path(package);
42 log::debug("writing to `{}`", archive_path.string());
43
44 std::ofstream archive(archive_path);
45 const auto [host, target] = util::net::parse_url(download_link);
46 const util::net::Requests requests{host};
47 std::ignore = requests.get(target, {}, std::move(archive));
48
49 return Ok(std::make_pair(archive_path, sha256sum));
50 } catch (const std::exception& e) {
51 return Result<std::pair<Path, String>>(Err<Unknown>(e.what()))
52 .with_context([&package] {
53 return Err<FailedToFetch>(package.name, package.version_rq).get();
54 });
55 } catch (...) {
56 return Err<FailedToFetch>(package.name, package.version_rq);
57 }
58}

63[[nodiscard]] Result<void>
64fetch(const UniqDeps<WithoutDeps>& deps) noexcept {
65 for (const auto& [name, version_rq] : deps) {
66 const resolve::Package package{name, version_rq};
67
68 const auto [installed_path, sha256sum] = Try(fetch_impl(package));
69 // Check if sha256sum of the downloaded package is the same with one
70 // stored in DB.
71 if (const String actual_sha256sum = Try(util::sha256::sum(installed_path));
72 sha256sum != actual_sha256sum) {
73 fs::remove(installed_path);
74 return Err<IncorrectSha256sum>(
75 sha256sum, actual_sha256sum, package.name, package.version_rq
76 );
77 }
78
79 const String extracted_directory_name =
80 Try(util::archive::extract(installed_path, config::path::extract_dir)
81 .map_err(to_anyhow));
82 Try(rename_extracted_directory(package, extracted_directory_name));
83
84 log::status("Downloaded", "{} v{}", package.name, package.version_rq);
85 }
86 return Ok();
87}

105[[nodiscard]] Result<void>
106download_deps(const ResolvedDeps& deps) noexcept {
107 const UniqDeps<WithoutDeps> not_installed_deps = get_not_installed_deps(deps);
108 if (not_installed_deps.empty()) {
109 // all resolved packages already have been installed
110 return Ok();
111 }
112
113 log::status("Downloading", "packages ...");
114 try {
115 fs::create_directories(config::path::cache_dir);
116 } catch (const std::exception& e) {
117 return Err<FailedToCreateDirs>(e.what());
118 }
119 Try(fetch(not_installed_deps));
120 return Ok();
121}

123[[nodiscard]] Result<ResolvedDeps>
124do_resolve(const UniqDeps<WithoutDeps>& deps) noexcept {
125 try {
126 const resolve::DupDeps<resolve::WithDeps> duplicate_deps =
127 Try(resolve::gather_all_deps(deps).map_err(to_anyhow));
128 if (!resolve::duplicate_loose(duplicate_deps)) {
129 // When all dependencies are composed of one package and one version,
130 // backtrack is not needed. Therefore, the `duplicate_loose`
131 // function just needs to check whether the gathered dependencies
132 // have multiple packages with the same name or not. If found multiple
133 // packages with the same name, then it means this package trying
134 // to be built depends on multiple versions of the same package.
135 // At the condition (the else clause), gathered dependencies
136 // should be in backtrack loop.
137 return Ok(ResolvedDeps(duplicate_deps.cbegin(), duplicate_deps.cend()));
138 } else {
139 return resolve::backtrack_loop(duplicate_deps).map_err(to_anyhow);
140 }
141 } catch (const std::exception& e) {
142 return Err<FailedToResolveDepsWithCause>(e.what());
143 } catch (...) {
144 return Err<FailedToResolveDeps>();
145 }
146}

148[[nodiscard]] Result<UniqDeps<WithoutDeps>>
149to_resolvable_deps(const HashMap<String, String>& dependencies) noexcept {
150 try {
151 // TOML tables should guarantee uniqueness.
152 UniqDeps<WithoutDeps> resolvable_deps{};
153 for (const auto& [name, version] : dependencies) {
154 resolvable_deps.emplace(name, version);
155 }
156 return Ok(resolvable_deps);
157 } catch (...) {
158 return Err<FailedToParseConfig>();
159 }

162[[nodiscard]] Result<ResolvedDeps>
163get_resolved_deps(const toml::value& manifest) {
164 auto deps = toml::find<HashMap<String, String>>(manifest, "dependencies");
165 if (manifest.contains("dev-dependencies")) {
166 append(
167 deps, toml::find<HashMap<String, String>>(manifest, "dev-dependencies")
168 );
169 }
170 const UniqDeps<WithoutDeps> resolvable_deps = Try(to_resolvable_deps(deps));
171 const ResolvedDeps resolved_deps = Try(do_resolve(resolvable_deps));
172 return Ok(resolved_deps);
173}

185[[nodiscard]] Result<ResolvedDeps>
186resolve_deps(const toml::value& manifest) {
187 const Option<ResolvedDeps> locked_deps = Try(try_to_read_lockfile());
188 if (locked_deps.has_value()) {
189 // Lockfile exists and is not outdated.
190 return Ok(locked_deps.value());
191 } else {
192 // Lockfile not found or outdated. Resolve dependencies from manifest file.
193 log::status("Resolving", "dependencies ...");
194 return get_resolved_deps(manifest);
195 }
196}

198[[nodiscard]] Result<ResolvedDeps>
199install_deps(const toml::value& manifest) {
200 if (!manifest.contains("dependencies")
201 && !manifest.contains("dev-dependencies")) {
202 const ResolvedDeps empty_deps{};
203 Try(data::lockfile::generate(empty_deps));
204 return Ok(empty_deps);
205 }
206
207 const ResolvedDeps resolved_deps = Try(resolve_deps(manifest));
208 Try(download_deps(resolved_deps));
209 Try(data::lockfile::generate(resolved_deps)); // when lockfile is old
210
211 return Ok(resolved_deps);
212}

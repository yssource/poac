70 inline SimpleResult
71 exec() const {
72 std::array<char, 128> buffer{};
73 std::string result;
74 FILE* pipe = popen(cmd.c_str(), "r");
75 if (!pipe) {
76 return {1, ""};
77 }
78 while (std::fgets(buffer.data(), 128, pipe) != nullptr) {
79 result += buffer.data();
80 }
81
82 const std::int32_t code = pclose(pipe);
83 if (code != 0) {
84 return {code, result};
85 }
86 return {0, result};
87 }

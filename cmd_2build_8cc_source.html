33[[nodiscard]] Result<Option<Path>>
34build(const Options& opts, const toml::value& manifest) {
35 const auto resolved_deps =
36 Try(core::resolver::install_deps(manifest).with_context([] {
37 return Err<FailedToInstallDeps>().get();
38 }));
39
40 // TODO(ken-matsui): We have to keep in mind a case of only dependencies
41 // require to be built, but this package does not.
42 if (!fs::exists(config::path::main_cpp_file)) {
43 log::status("Finished", "no build target(s) found");
44 return Ok(None);
45 }
46
47 const Mode mode = opts.release.value() ? Mode::release : Mode::debug;
48 const Path output_path = Try(build_impl(manifest, mode, resolved_deps));
49 return Ok(output_path);
50}

52[[nodiscard]] Result<void>
53exec(const Options& opts) {
54 spdlog::trace("Checking if required config exists ...");
55 Try(core::validator::required_config_exists().map_err(to_anyhow));
56
57 spdlog::trace("Parsing the manifest file ...");
58 // TODO(ken-matsui): parse as a static type rather than toml::value
59 const toml::value manifest = toml::parse(data::manifest::name);
60
61 Try(build(opts, manifest).with_context([&manifest] {
62 return Err<FailedToBuild>(toml::find<String>(manifest, "package", "name"))
63 .get();
64 }));
65 return Ok();

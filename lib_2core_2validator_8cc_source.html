79[[nodiscard]] Result<void, String>
80invalid_characters_impl(StringRef s) noexcept {
81 for (const char c : s) {
82 if (!is_alpha_numeric(c) && c != '_' && c != '-' && c != '/') {
83 return Err(
84 "Invalid package name.\n"
85 "It is prohibited to use a string\n"
86 " that does not match ^([a-z|\\d|_|\\-|\\/]*)$."

251[[nodiscard]] Result<void, String>
252valid_athr(StringRef s) {
253 // TODO(ken-matsui): Email address parser
254 if (usize pos = s.find('<'); pos != SNone) {
255 if (pos = s.find('@', pos + 1); pos != SNone) {
256 if (pos = s.find('>', pos + 1); pos != SNone) {
257 if (s.ends_with('>')) {
258 return Ok();
259 }
260 }
261 }
262 }
263 return Err(format(
264 "author `{}` is written in invalid style. It should be like:\n"
265 " `Your Name <your-public@email.address>`",

317[[nodiscard]] Result<void, String>
318valid_repository(StringRef repo) {
319 // Can be parsed? it should be:
320 // https://github.com/org/repo/tree/tag
321 if (repo.starts_with("https://github.com/")) {
322 // org/
323 // 19: size of `https://github.com/`
324 if (usize pos = repo.find('/', 19); pos != SNone) {
325 // repo/
326 if (pos = repo.find('/', pos + 1); pos != SNone) {
327 // tree/
328 if (repo.substr(pos + 1, 5) == "tree/") {
329 // repo/tree/
330 // ^---->^
331 pos += 5;
332 // tag
333 if (repo.size() - 1 > pos) {
334 // -- tag found case
335 // tree/tag `8 (size)`
336 // ^ `5 (pos)`
337
338 // -- no tag case
339 // tree/ `5 (size)`
340 // ^ `5 (pos)`
341 return Ok();
342 }
343 }
344 }
345 }
346 }
347 return Err(format(
348 "`{}` is invalid form for the `repository` key. It should be like:\n"
349 " `https://github.com/org/repo/tree/tag`",
350 repo
351 ));
352}

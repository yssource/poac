56[[nodiscard]] Result<void>
57create(const Options& opts) {
58 std::ofstream ofs;
59 const ProjectType type = opts_to_project_type(opts);
60 for (auto&& [name, text] : create_template_files(type, opts.package_name)) {
61 const String& file_path = (opts.package_name / name).string();
62 spdlog::trace("Creating {}", file_path);
63 write_to_file(ofs, file_path, text);
64 }
65
66 spdlog::trace("Initializing git repository at {}", opts.package_name);
67 git2::repository().init(opts.package_name);
68
69 log::status("Created", "{} `{}` package", to_string(type), opts.package_name);
70 return Ok();
71}

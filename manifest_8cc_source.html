19bool
20rebuild(data::NinjaMain& ninja_main, Status& status, String& err) {
21 Node* node = ninja_main.state.LookupNode(
22 (ninja_main.build_dir / manifest_file_name).string()
23 );
24 if (!node) {
25 return false;
26 }
27
28 Builder builder(
29 &ninja_main.state, ninja_main.config, &ninja_main.build_log,
30 &ninja_main.deps_log, &ninja_main.disk_interface, &status,
31 ninja_main.start_time_millis
32 );
33 if (!builder.AddTarget(node, &err)) {
34 return false;
35 }
36 if (builder.AlreadyUpToDate()) {
37 return false; // Not an error, but we didn't rebuild.
38 }
39 if (!builder.Build(&err)) {
40 return false;
41 }
42
43 // The manifest was only rebuilt if it is now dirty (it may have been cleaned
44 // by a restat).
45 if (!node->dirty()) {
46 // Reset the state to prevent problems like
47 // https://github.com/ninja-build/ninja/issues/874
48 ninja_main.state.Reset();
49 return false;
50 }
51 return true;
52}

54Vec<String>
55gather_includes(const resolver::ResolvedDeps& resolved_deps) {
56 Vec<String> includes;
57 for (const auto& [package, inner_deps] : resolved_deps) {
58 static_cast<void>(inner_deps);
59
60 const Path include_path = resolver::get_extracted_path(package) / "include";
61 if (fs::exists(include_path) && fs::is_directory(include_path)) {
62 includes.emplace_back(format("-I{}", include_path.string()));
63 }
64 }
65 return includes;
66}

68Vec<toml::table>
69get_cfg_profile(const toml::value& poac_manifest) {
70 const auto target =
71 toml::find_or<toml::table>(poac_manifest, "target", toml::table{});
72 Vec<toml::table> profiles;
73 for (const auto& [key, val] : target) {
74 if (key.find("cfg(") != SNone) {
75 if (util::cfg::parse(key).match()) {
76 const auto profile =
77 toml::find_or<toml::table>(val, "profile", toml::table{});
78 profiles.emplace_back(profile);
79 }
80 }
81 }
82 return profiles;
83}

102[[nodiscard]] Result<String>
103construct(
104 const Path& build_dir, const toml::value& poac_manifest,
105 const resolver::ResolvedDeps& resolved_deps
106) {
107 syntax::Writer writer{std::ostringstream()};
108 for (StringRef header : manifest_headers) {
109 writer.comment(String(header));
110 }
111 writer.newline();
112
113 const String name = toml::find<String>(poac_manifest, "package", "name");
114 const String version =
115 toml::find<String>(poac_manifest, "package", "version");
116 const i64 edition = toml::find<i64>(poac_manifest, "package", "edition");
117 const String command = Try(compiler::cxx::get_command(edition, false));
118
119 writer.rule(
120 "compile",
121 format("{} $OPTIONS $DEFINES $INCLUDES $LIBRARIES $in -o $out", command),
122 syntax::RuleSet{
123 .description = "$PACKAGE_NAME v$PACKAGE_VERSION $PACKAGE_PATH",
124 }
125 );
126 writer.newline();
127
128 const Path source_file = "src"_path / "main.cpp";
129 Path output_file;
130 if (source_file == "src"_path / "main.cpp") {
131 // When building src/main.cpp, the output executable should be stored at
132 // poac_output/debug/name
133 output_file = build_dir / name;
134 } else {
135 output_file = (build_dir / source_file).string() + ".o";
136 fs::create_directories(output_file.parent_path());
137 }
138 const Vec<String> includes = gather_includes(resolved_deps);
139
140 const Vec<String> defines = gather_flags(poac_manifest, "definitions", "-D");
141 const Vec<String> options = gather_flags(poac_manifest, "options");
142 const Vec<String> libraries = gather_flags(poac_manifest, "libraries", "-l");
143
144 writer.build(
145 {output_file.string()}, "compile",
146 syntax::BuildSet{
147 .inputs = std::vector{source_file.string()},
148 .variables =
149 syntax::Variables{
150 {"PACKAGE_NAME", name},
151 {"PACKAGE_VERSION", version},
152 {"PACKAGE_PATH", format("({})", config::path::cur_dir)},
153 {"OPTIONS", boost::algorithm::join(options, " ")},
154 {"DEFINES", boost::algorithm::join(defines, " ")},
155 {"INCLUDES", boost::algorithm::join(includes, " ")},
156 {"LIBRARIES", boost::algorithm::join(libraries, " ")},
157 },
158 }
159 );
160 writer.newline();
161
162 writer.default_({output_file.string()});
163 return Ok(writer.get_value());
164}

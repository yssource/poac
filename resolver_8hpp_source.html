63inline String
64convert_to_download_link(StringRef repository) {
65 // repository should be like =>
66 // https://github.com/boostorg/winapi/tree/boost-1.66.0
67 // convert it to =>
68 // https://github.com/boostorg/winapi/archive/boost-1.66.0.tar.gz
69 //
70 // We should prevent being broken with a package name used `tree` here.
71 // Correct:
72 // https://github.com/tree/tree/tree/tree/v0.1.0
73 // => https://github.com/tree/tree/archive/tree/v0.1.0
74
75 // The first `tree/` is organization name.
76 usize start =
77 repository.find('/', 19) + 1; // 19: size of `https://github.com/`
78 // The next `tree/` is repository name.
79 start = repository.find('/', start) + 1;
80 // `start` is now pointing `t` after the repository name.
81 // So, find the end of `tree/`.
82 const usize end = repository.find('/', start);
83 // Retrieve both sides: `https://github.com/tree/tree/`
84 StringRef left = repository.substr(0, start);
85 // `/tree/v0.1.0`: this side is just a tag.
86 // Mostly, we do not include `tree`, but we can.
87 StringRef right = repository.substr(end);
88 return format("{}archive{}.tar.gz", left, right);
89}

110[[nodiscard]] Result<void>
111set_color_mode(StringRef color_mode) {
112 if (color_mode == "auto") {
113 termcolor2::set_color_mode(spdlog::color_mode::automatic);
114 if (termcolor2::should_color()) {
115 toml::color::enable();
116 } else {
117 toml::color::disable();
118 }
119 } else if (color_mode == "always") {
120 termcolor2::set_color_mode(spdlog::color_mode::always);
121 toml::color::enable();
122 } else if (color_mode == "never") {
123 termcolor2::set_color_mode(spdlog::color_mode::never);
124 toml::color::disable();
125 } else {
126 return Err<ColorError>(color_mode);
127 }
128 return Ok();
129}

161int
162main(const int argc, char* argv[]) {
163 spdlog::set_pattern("%v");
164 auto err_logger = spdlog::stderr_logger_st("stderr");
165
166 auto app = structopt::app("poac", POAC_VERSION);
167 try {
168 const auto args = app.parse<Commands>(argc, argv);
169
170 // Global options
171 if (args.verbose.value()) {
172 spdlog::set_level(spdlog::level::trace);
173 } else if (args.quiet.value()) {
174 spdlog::set_level(spdlog::level::off);
175 }
176
177 // Subcommands
178 return exec(app, args)
179 .map_err([err_logger](const auto& e) {
180 log::error(
181 err_logger, colorize_anyhow_error(format("{}", e->what()))
182 );
183 })
184 .is_err();
185 } catch (const structopt::exception& e) {
186 if (argc > 1) {
187 // try correcting typo
188 if (const auto sugg =
189 util::lev_distance::find_similar_str(argv[1], command_list)) {
190 err_logger->error(
191 "{}\n"
192 " --> Did you mean `{}`?\n\n"
193 "For more information, try {}",
194 colorize_structopt_error(e.what()), sugg.value(), "--help"_green
195 );
196 return EXIT_FAILURE;
197 }
198 }
199 err_logger->error(
200 "{}\n\nFor more information, try {}",
201 colorize_structopt_error(e.what()), "--help"_green
202 );
203 return EXIT_FAILURE;
204 } catch (const std::exception& e) {
205 log::error(err_logger, e.what());
206 return EXIT_FAILURE;
207 } catch (...) {
208 err_logger->error(
209 "{} Unknown error occurred\n\n"
210 "Please open an issue with reproducible information at:\n"
211 "https://github.com/poacpm/poac/issues",
212 "Error:"_bold_red
213 );
214 return EXIT_FAILURE;
215 }

16 const auto exp = [&](const boost::smatch& m) {

52void
53Writer::_line(String text, usize indent) {
54 String leading_space = String(" ") * indent;
55
56 while (leading_space.length() + text.length() > width) {
57 // The text is too wide; wrap if possible.
58
59 // Find the rightmost space that would obey our width constraint and
60 // that's not an escaped space.
61 std::int32_t available_space =
62 width - leading_space.length() - 2; // " $".length() == 2
63 std::int32_t space = available_space;
64 do {
65 space = text.rfind(' ', space);
66 } while (!(space < 0 || count_dollars_before_index(text, space) % 2 == 0));
67
68 if (space < 0) {
69 // No such space; just use the first unescaped space we can find.
70 space = available_space - 1;
71 do {
72 space = text.find(' ', space + 1);
73 } while (!(space < 0 || count_dollars_before_index(text, space) % 2 == 0)
74 );
75 }
76 if (space < 0) {
77 // Give up on breaking.
78 break;
79 }
80
81 output << leading_space + text.substr(0, space) + " $\n";
82 text = text.substr(space + 1);
83
84 // Subsequent lines are continuations, so indent them.
85 leading_space = String(" ") * (indent + 2);
86 }
87 output << leading_space + text + '\n';
88}

136Writer::build(

164 all_inputs.emplace_back("||");
165 boost::push_back(all_inputs, order_only);
166 }
167 if (build_set.implicit_outputs.has_value()) {
168 Vec<String> implicit_outputs;
169 for (const Path& i : build_set.implicit_outputs.value()) {
170 implicit_outputs.emplace_back(escape_path(i).string());
171 }
172 out_outputs.emplace_back("|");
173 boost::push_back(out_outputs, implicit_outputs);
174 }
175
176 _line(format(
177 "build {}: {} {}", boost::algorithm::join(out_outputs, " "), rule,
178 boost::algorithm::join(all_inputs, " ")

182 _line(format(" pool = {}", build_set.pool.value()));
183 }
184 if (build_set.dyndep.has_value()) {
185 _line(format(" dyndep = {}", build_set.dyndep.value()));
186 }
187
188 if (build_set.variables.has_value()) {
189 for (const auto& [key, val] : build_set.variables.value()) {
190 variable(key, val, 1);
191 }
192 }

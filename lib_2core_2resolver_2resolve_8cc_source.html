23Vec<Vec<i32>>
24multiple_versions_cnf(const Vec<i32>& clause) {
25 return boost::irange(0, 1 << clause.size()) // number of combinations
26 | boost::adaptors::transformed([&clause](const i32 i) {
27 return boost::dynamic_bitset<>(to_binary_numbers(i, clause.size())
28 );
29 })
30 | boost::adaptors::filtered([](const boost::dynamic_bitset<>& bs) {
31 return bs.count() != 1;
32 })
33 | boost::adaptors::transformed(
34 [&clause](const boost::dynamic_bitset<>& bs) -> Vec<i32> {
35 return boost::irange(usize{0}, bs.size())
36 | boost::adaptors::transformed([&clause,
37 &bs](const i32 i) {
38 return bs[i] ? clause[i] * -1 : clause[i];
39 })
40 | util::meta::containerized;
41 }
42 )
43 | util::meta::containerized;
44}

46Vec<Vec<i32>>
47create_cnf(const DupDeps<WithDeps>& activated) {
48 Vec<Vec<i32>> clauses;
49 Vec<i32> already_added;
50
51 DupDeps<WithDeps>::const_iterator first = std::cbegin(activated);
52 DupDeps<WithDeps>::const_iterator last = std::cend(activated);
53 for (usize i = 0; i < activated.size(); ++i) {
54 if (util::meta::find(already_added, i)) {
55 continue;
56 }
57
58 const auto name_lambda = [&](const auto& x) {
59 return get_package(x) == get_package(activated[i]);
60 };
61 // No other packages with the same name as the package currently pointed to
62 // exist
63 if (const i64 count = std::count_if(first, last, name_lambda); count == 1) {
64 Vec<i32> clause;
65 clause.emplace_back(i + 1);
66 clauses.emplace_back(clause);
67
68 // index ⇒ deps
69 if (!activated[i].second.has_value()) {
70 clause[0] *= -1;
71 for (const auto& [name, version] : activated[i].second.value()) {
72 // It is guaranteed to exist
73 clause.emplace_back(
74 util::meta::index_of_if(
75 first, last,
76 [&n = name, &v = version](const auto& d) {
77 return get_package(d).name == n
78 && get_package(d).version_rq == v;
79 }
80 )
81 + 1
82 );
83 }
84 clauses.emplace_back(clause);
85 }
86 } else if (count > 1) {
87 Vec<i32> clause;
88
89 for (DupDeps<WithDeps>::const_iterator found = first; found != last;
90 found = std::find_if(found, last, name_lambda)) {
91 const i64 index = std::distance(first, found);
92 clause.emplace_back(index + 1);
93 already_added.emplace_back(index + 1);
94
95 // index ⇒ deps
96 if (!found->second.has_value()) {
97 Vec<i32> new_clause;
98 new_clause.emplace_back(index);
99 for (const Package& package : found->second.value()) {
100 // It is guaranteed to exist
101 new_clause.emplace_back(
102 util::meta::index_of_if(
103 first, last,
104 [&package](const auto& p) {
105 return get_package(p).name == package.name
106 && get_package(p).version_rq == package.version_rq;
107 }
108 )
109 + 1
110 );
111 }
112 clauses.emplace_back(new_clause);
113 }
114 ++found;
115 }
116 boost::range::push_back(clauses, multiple_versions_cnf(clause));
117 }
118 }
119 return clauses;
120}

122[[nodiscard]] Result<UniqDeps<WithDeps>, String>
123solve_sat(const DupDeps<WithDeps>& activated, const Vec<Vec<i32>>& clauses) {
124 // deps.activated.size() == variables
125 const Vec<i32> assignments = Try(sat::solve(clauses, activated.size()));
126 UniqDeps<WithDeps> resolved_deps{};
127 log::debug("SAT");
128 for (i32 a : assignments) {
129 log::debug("{} ", a);
130 if (a > 0) {
131 const auto& [package, deps] = activated[a - 1];
132 resolved_deps.emplace(package, deps);
133 }
134 }
135 log::debug(0);
136 return Ok(resolved_deps);
137}

139[[nodiscard]] Result<UniqDeps<WithDeps>, String>
140backtrack_loop(const DupDeps<WithDeps>& activated) {
141 const Vec<Vec<i32>> clauses = create_cnf(activated);
142 if (util::verbosity::is_verbose()) {
143 for (const Vec<i32>& c : clauses) {
144 for (i32 l : c) {
145 const auto deps = activated[std::abs(l) - 1];
146 const Package package = get_package(deps);
147 log::debug("{}-{}: {}, ", package.name, package.version_rq, l);
148 }
149 log::debug("");
150 }
151 }
152 return solve_sat(activated, clauses);
153}

158[[nodiscard]] Result<Vec<String>, String>
159get_versions_satisfy_interval(const Package& package) {
160 // TODO(ken-matsui): (`>1.2 and <=1.3.2` -> NG，`>1.2.0-alpha and <=1.3.2` ->
161 // OK) `2.0.0` specific version or `>=0.1.2 and <3.4.0` version interval
162 const semver::Interval i(package.version_rq);
163 const Vec<String> satisfied_versions =
164 Try(util::net::api::versions(package.name))
165 | boost::adaptors::filtered([&i](StringRef s) { return i.satisfies(s); })
166 | util::meta::containerized;
167
168 if (satisfied_versions.empty()) {
169 return Err(format(
170 "`{}: {}` not found; seem dependencies are broken", package.name,
171 package.version_rq
172 ));
173 }
174 return Ok(satisfied_versions);
175}

177DupDeps<WithoutDeps>
178gather_deps_of_deps(
179 const UniqDeps<WithoutDeps>& deps_api_res, IntervalCache& interval_cache
180) {
181 DupDeps<WithoutDeps> cur_deps_deps;
182 for (const auto& [name, version_rq] : deps_api_res) {
183 const Package package{name, version_rq};
184
185 // Check if node package is resolved dependency (by interval)
186 const IntervalCache::iterator found_cache =
187 boost::range::find_if(interval_cache, [&package](const Cache& cache) {
188 return package == cache.package;
189 });
190
191 const Vec<String> dep_versions =
192 found_cache != interval_cache.cend()
193 ? found_cache->versions
194 : get_versions_satisfy_interval(package).unwrap();
195 if (found_cache == interval_cache.cend()) {
196 // Cache interval and versions pair
197 interval_cache.emplace(Cache{package, dep_versions});
198 }
199 for (const String& dep_version : dep_versions) {
200 cur_deps_deps.emplace_back(Package{package.name, dep_version});
201 }
202 }
203 return cur_deps_deps;
204}

206void
207gather_deps(
208 const Package& package, DupDeps<WithDeps>& new_deps,
209 IntervalCache& interval_cache
210) {
211 // Check if root package resolved dependency (whether the specific version is
212 // the same), and check circulating
213 if (cache_exists(new_deps, package)) {
214 return;
215 }
216
217 // Get dependencies of dependencies
218 const UniqDeps<WithoutDeps> deps_api_res =
219 util::net::api::deps(package.name, package.version_rq).unwrap();
220 if (deps_api_res.empty()) {
221 new_deps.emplace_back(package, None);
222 } else {
223 const DupDeps<WithoutDeps> deps_of_deps =
224 gather_deps_of_deps(deps_api_res, interval_cache);
225
226 // Store dependency and the dependency's dependencies.
227 new_deps.emplace_back(package, deps_of_deps);
228
229 // Gather dependencies of dependencies of dependencies.
230 for (const Package& dep_package : deps_of_deps) {
231 gather_deps(dep_package, new_deps, interval_cache);
232 }
233 }

236[[nodiscard]] Result<DupDeps<WithDeps>, String>
237gather_all_deps(const UniqDeps<WithoutDeps>& deps) {
238 DupDeps<WithDeps> duplicate_deps;
239 IntervalCache interval_cache;
240
241 // Activate the root of dependencies
242 for (const auto& [name, version_rq] : deps) {
243 const Package package{name, version_rq};
244
245 // Check whether the packages specified in poac.toml
246 // are already resolved which includes
247 // that package's dependencies and package's versions
248 // by checking whether package's interval is the same.
249 if (cache_exists(interval_cache, package)) {
250 continue;
251 }
252
253 // Get versions using interval
254 // FIXME: versions API and deps API are received the almost same responses
255 const Vec<String> versions = Try(get_versions_satisfy_interval(package));
256 // Cache interval and versions pair
257 interval_cache.emplace(Cache{package, versions});
258 for (const String& version : versions) {
259 gather_deps(
260 Package{package.name, version}, duplicate_deps, interval_cache
261 );
262 }
263 }
264 return Ok(duplicate_deps);
265}

6// Create progress bar, [====> ]
7String
8to_progress(const i32& max_count, i32 now_count, const i32& bar_size) {
9 if (now_count > max_count) {
10 now_count = max_count;
11 }
12 const i32 percent = (now_count * 100) / max_count;
13 const i32 bar_pos = percent / 2;
14
15 if (now_count == max_count) {
16 return format("[{:=>{}}", ">]", bar_size + 1);
17 } else if ((bar_pos - 1) > 0) {
18 return format("[{:=>{}}{:>{}}", ">", bar_pos, "]", bar_size - bar_pos + 1);
19 } else if (bar_pos == 1) {
20 return format("[>{:>{}}", "]", bar_size);
21 } else {
22 return format("[{:>{}}", "]", bar_size + 1);
23 }
24}

48void
49MultiPartForm::generate_header() {
50 m_header = format("{}{}", m_crlf, fmt::join(m_form_param, ""));
51 for (const auto& [name, filename, header] : m_file_param) {
52 String h = format(
53 "--{}{}{}name=\"{}\"; filename=\"{}\"", m_boundary, m_crlf,
54 m_content_disposition, name, filename.filename().string()
55 );
56 for (const auto& [field, content] : header) {
57 h += format("{}{}: {}", m_crlf, String(to_string(field)), content);
58 }
59 m_header += m_crlf + h;
60 }
61 m_header += m_crlf + m_crlf;
62}

96[[nodiscard]] Result<boost::property_tree::ptree, String>
97call(StringRef path, StringRef body) noexcept {
98 try {
99 const Requests request{
100 format("{}.functions.supabase.co", SUPABASE_PROJECT_REF)};
101 Headers headers;
102 headers.emplace(
103 boost::beast::http::field::authorization,
104 format("Bearer {}", SUPABASE_ANON_KEY)
105 );
106
107 const auto response = Try(request.post(path, body, headers));
108 std::stringstream response_body;
109 response_body << response.data();
110
111 boost::property_tree::ptree pt;
112 boost::property_tree::json_parser::read_json(response_body, pt);
113 return Ok(pt);
114 } catch (const std::exception& e) {
115 return Err(e.what());
116 } catch (...) {
117 return Err("unknown error caused when calling search api");
118 }
119}

27Result<std::pair<Graph, Vec<String>>>
28create_graph() {
29 const core::resolver::ResolvedDeps resolved_deps =
30 Try(create_resolved_deps());
31 Graph g;
32
33 // Add vertex
34 Vec<Graph::vertex_descriptor> desc;
35 for (const auto& dep : resolved_deps | boost::adaptors::indexed()) {
36 desc.push_back(boost::add_vertex(g));
37
38 const i64 index = dep.index();
39 const core::resolver::resolve::Package package = dep.value().first;
40
41 g[index].name = package.name;
42 g[index].version = package.version_rq;
43 }
44
45 // Add edge
46 for (const auto& dep : resolved_deps | boost::adaptors::indexed()) {
47 const i64 index = dep.index();
48 const auto deps = dep.value().second;
49
50 if (deps.has_value()) {
51 for (const auto& [name, version] : deps.value()) {
52 auto first = resolved_deps.cbegin();
53 auto last = resolved_deps.cend();
54
55 const auto result = std::find_if(first, last, [&n = name](auto d) {
56 // dependency should be resolved as only one package.
57 return d.first.name == n;
58 });
59 if (result != last) {
60 boost::add_edge(desc[index], desc[std::distance(first, result)], g);
61 }
62 }
63 }
64 }
65
66 Vec<String> names;
67 for (const auto& [package, deps] : resolved_deps) {
68 static_cast<void>(deps);
69 names.push_back(package.name + ": " + package.version_rq);
70 }
71 return Ok(std::make_pair(g, names));
72}

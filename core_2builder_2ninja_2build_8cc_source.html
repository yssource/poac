45[[nodiscard]] Result<void>
46run(data::NinjaMain& ninja_main, Status& status) {
47 String err;
48 Vec<Node*> targets = ninja_main.state.DefaultNodes(&err);
49 if (!err.empty()) {
50 return Err<GeneralError>(err);
51 }
52 ninja_main.disk_interface.AllowStatCache(true);
53
54 Builder builder(
55 &ninja_main.state, ninja_main.config, &ninja_main.build_log,
56 &ninja_main.deps_log, &ninja_main.disk_interface, &status,
57 ninja_main.start_time_millis
58 );
59 for (usize i = 0; i < targets.size(); ++i) {
60 if (!builder.AddTarget(targets[i], &err)) {
61 if (!err.empty()) {
62 return Err<GeneralError>(err);
63 }
64 // Added a target that is already up-to-date; not really an error.
65 }
66 }
67 // Make sure restat rules do not see stale timestamps.
68 ninja_main.disk_interface.AllowStatCache(false);
69
70 if (builder.AlreadyUpToDate()) {
71 spdlog::trace("nothing to do.");
72 return Ok();
73 }
74 if (!builder.Build(&err)) {
75 return Err<GeneralError>(err);
76 }
77 return Ok();
78}

91[[nodiscard]] Result<Path>
92start(
93 const toml::value& poac_manifest, const Mode& mode,
94 const resolver::ResolvedDeps& resolved_deps
95) {
96 BuildConfig config;
97
98 // ref: https://github.com/ninja-build/ninja/pull/2102#issuecomment-1147771497
99 setenv("TERM", "dumb", true);
100 const String progress_status_format =
101 termcolor2::should_color()
102 ? format("{:>27} %f/%t: ", "Compiling"_bold_green)
103 : format("{:>12} %f/%t: ", "Compiling");
104 setenv("NINJA_STATUS", progress_status_format.c_str(), true);
105 StatusPrinter status(config);
106
107 const Path build_dir = config::path::output_dir / to_string(mode);
108 fs::create_directories(build_dir);
109 Try(manifest::create(build_dir, poac_manifest, resolved_deps));
110
111 for (i32 cycle = 1; cycle <= rebuildLimit; ++cycle) {
112 data::NinjaMain ninja_main(config, build_dir);
113 ManifestParserOptions parser_opts;
114 parser_opts.dupe_edge_action_ = kDupeEdgeActionError;
115 ManifestParser parser(
116 &ninja_main.state, &ninja_main.disk_interface, parser_opts
117 );
118 String err;
119 if (!parser.Load(
120 (ninja_main.build_dir / manifest::manifest_file_name).string(), &err
121 )) {
122 return Err<GeneralError>(err);
123 }
124
125 Try(log::load_build_log(ninja_main));
126 Try(log::load_deps_log(ninja_main));
127
128 // Attempt to rebuild the manifest before building anything else
129 if (manifest::rebuild(ninja_main, status, err)) {
130 // Start the build over with the new manifest.
131 continue;
132 } else if (!err.empty()) {
133 return Err<GeneralError>(err);
134 }
135
136 Try(run(ninja_main, status));
137 return Ok(config::path::output_dir / to_string(mode));
138 }
139 return Err<GeneralError>(format(
140 "internal manifest still dirty after {} tries, perhaps system time is not set",
141 rebuildLimit
142 ));
143}

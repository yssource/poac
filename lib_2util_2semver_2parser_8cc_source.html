109std::vector<Identifier>
110Parser::pre() {
111 if (const Token p = peek(); p.kind == Token::Whitespace) {
112 pop(); // Drop whitespace
113 if (const Token p2 = peek();
114 p2 != Token::Unexpected && !p2.is_whitespace()) {
115 // `1.2.3 a.b.c`
116 throw version_error(
117 "continuing pre-release identifiers after spaces is not allowed"
118 );
119 }
120 return {};
121 } else if (p != Token::Hyphen) {
122 return {};
123 }

125 pop();
126 return parts();

159Op
160Parser::op() {
161 Op op(Op::Compatible);
162
163 switch (peek().kind) {
164 case Token::Eq:
165 op.kind = Op::Ex;
166 break;
167 case Token::Gt:
168 op.kind = Op::Gt;
169 break;
170 case Token::GtEq:
171 op.kind = Op::GtEq;
172 break;
173 case Token::Lt:
174 op.kind = Op::Lt;
175 break;
176 case Token::LtEq:
177 op.kind = Op::LtEq;
178 break;
179 case Token::Tilde:
180 op.kind = Op::Tilde;
181 break;
182 case Token::Caret:
183 op.kind = Op::Compatible;
184 break;
185 // default op
186 default:
187 op.kind = Op::Compatible;
188 break;
189 }
190 // remove the matched token.
191 pop();
192 skip_whitespace();
193 return op;
194}

199std::optional<Predicate>
200Parser::predicate() {
201 if (is_eof()) {
202 return std::nullopt;
203 }
204
205 Op op = this->op();
206
207 std::uint_fast64_t major;
208 if (const auto m = this->component()) {
209 major = m.value();
210 } else {
211 return std::nullopt;
212 }
213
214 const auto minor = this->dot_component();
215 const auto patch = this->dot_component();
216 const auto pre = this->pre();
217
218 // TODO(ken-matsui): avoid illegal combinations, like `1.*.0`.
219 if (!minor.has_value()) {
220 op = Op(Op::Wildcard, WildcardVersion::Minor);
221 }
222 if (!patch.has_value()) {
223 op = Op(Op::Wildcard, WildcardVersion::Patch);
224 }
225
226 // ignore build metadata
227 this->plus_build_metadata();
228
229 return Predicate{op, major, minor, patch, pre};
230}

295bool
296Parser::has_ws_separator(const Token::Kind& pat) {
297 skip_whitespace();
298
299 if (peek() == pat) {

301 pop();

10[[nodiscard]] Result<toml::basic_value<toml::preserve_comments>>
11convert_to_lock(const resolver::UniqDeps<resolver::WithDeps>& deps) {
12 Vec<Package> packages;
13 for (const auto& [pack, inner_deps] : deps) {
14 Package p{
15 pack.name,
16 pack.version_rq,
17 Vec<String>{},
18 };
19 if (inner_deps.has_value()) {
20 // Extract name from inner dependencies and drop version.
21 Vec<String> ideps;
22 for (const auto& [name, _v] : inner_deps.value()) {
23 static_cast<void>(_v);
24 ideps.emplace_back(name);
25 }
26 p.dependencies = ideps;
27 }
28 packages.emplace_back(p);
29 }
30
31 toml::basic_value<toml::preserve_comments> lock(
32 Lockfile{.package = packages}, {String(lockfile_header)}
33 );
34 return Ok(lock);
35}

55[[nodiscard]] resolver::UniqDeps<resolver::WithDeps>
56convert_to_deps(const Lockfile& lock) {
57 resolver::UniqDeps<resolver::WithDeps> deps;
58 for (const auto& package : lock.package) {
59 resolver::UniqDeps<resolver::WithDeps>::mapped_type inner_deps = None;
60 if (!package.dependencies.empty()) {
61 // When serializing lockfile, package version of inner dependencies
62 // will be dropped (ref: `convert_to_lock` function).
63 // Thus, the version should be restored just as empty string ("").
64 resolver::UniqDeps<resolver::WithDeps>::mapped_type::value_type ideps;
65 for (const auto& name : package.dependencies) {
66 ideps.push_back({name, ""});
67 }
68 inner_deps = ideps;
69 }
70 deps.emplace(resolver::Package{package.name, package.version}, inner_deps);
71 }
72 return deps;
73}

75[[nodiscard]] Result<Option<resolver::UniqDeps<resolver::WithDeps>>>
76read(const Path& base_dir) {
77 if (!fs::exists(base_dir / lockfile_name)) {
78 return Ok(None);
79 }
80
81 try {
82 const toml::value lock = toml::parse(base_dir / lockfile_name);
83 const Lockfile parsed_lock = toml::get<Lockfile>(lock);
84 if (parsed_lock.version != lockfile_version) {
85 return Err<InvalidLockfileVersion>(parsed_lock.version);
86 }
87 return Ok(convert_to_deps(parsed_lock));
88 } catch (const std::exception& e) {
89 return Err<FailedToReadLockfile>(e.what());
90 }
91}

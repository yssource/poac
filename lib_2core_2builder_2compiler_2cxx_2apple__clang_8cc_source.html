9[[nodiscard]] Result<semver::Version>
10get_compiler_version_impl(const String& cmd_output) {
11 // `Apple clang version 13.0.0 (...)`
12 String version;
13 for (usize i = 20; i < cmd_output.size(); ++i) {
14 if (cmd_output[i] == ' ') { // read until space
15 break;
16 }
17 version += cmd_output[i];
18 }
19 return Ok(semver::parse(version));
20}

22[[nodiscard]] Result<semver::Version>
23get_compiler_version(const String& compiler_command) {
24 const auto res =
25 util::shell::Cmd(compiler_command + " --version").dump_stderr().exec();
26 if (res.is_ok()) {
27 return get_compiler_version_impl(res.output());

29 return Err<error::FailedToGetCompilerVersion>(compiler);
30}

34[[nodiscard]] Result<String>
35get_std_flag(
36 const String& compiler_command, const i64 edition,
37 const bool use_gnu_extension
38) {
39 const semver::Version version = Try(get_compiler_version(compiler_command));
40 const String specifier = use_gnu_extension ? "gnu" : "c";
41 switch (edition) {
42 case 1998:
43 if (version >= "4.0.0") {
44 return Ok(format("-std={}++98", specifier));
45 }
46 break;
47 case 2011:
48 if (version >= "4.0.0") {
49 return Ok(format("-std={}++11", specifier));
50 }
51 break;
52 case 2014:
53 if (version >= "6.1.0") {
54 return Ok(format("-std={}++14", specifier));
55 } else if (version >= "5.1.0") {
56 // AppleClang 5.0 knows this flag, but does not set a __cplusplus macro
57 // greater than 201103L
58 return Ok(format("-std={}++1y", specifier));
59 }
60 break;
61 case 2017:
62 if (version >= "10.0.0") {
63 return Ok(format("-std={}++17", specifier));
64 } else if (version >= "6.1.0") {
65 return Ok(format("-std={}++1z", specifier));
66 }
67 break;
68 case 2020:
69 if (version >= "13.0.0") {
70 return Ok(format("-std={}++20", specifier));
71 } else if (version >= "10.0.0") {
72 return Ok(format("-std={}++2a", specifier));
73 }
74 break;
75 }
76 return Err<error::UnsupportedLangVersion>(
77 compiler, version, lang::Lang::cxx, edition
78 );
79}

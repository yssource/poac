148// The following Regular Expressions can be used for tokenizing,
149// validating, and parsing SemVer version strings.
150// A regular expression before binding is
151// https://github.com/semver/semver/issues/232#issue-48635632
152
153// ## Numeric Identifier
154// A single `0`, or a non-zero digit followed by zero or more digits.
155inline const std::string NUMERIC_IDENTIFIER = R"(0|[1-9]\d*)";
156
157// ## Non-numeric Identifier
158// Zero or more digits, followed by a letter or hyphen, and then zero or
159// more letters, digits, or hyphens.
160inline const std::string NON_NUMERIC_IDENTIFIER =
161 R"(\d*[a-zA-Z-][0-9a-zA-Z-]*)";
162
163// ## Main Version
164// Three dot-separated numeric identifiers.
165inline const std::string MAIN_VERSION = "(" + NUMERIC_IDENTIFIER + R"()\.)"
166 + "(" + NUMERIC_IDENTIFIER + R"()\.)"
167 + "(" + NUMERIC_IDENTIFIER + ")";
168
169// ## Pre-release Version Identifier
170// A numeric identifier, or a non-numeric identifier.
171inline const std::string PRE_RELEASE_IDENTIFIER =
172 "(?:" + NUMERIC_IDENTIFIER + "|" + NON_NUMERIC_IDENTIFIER + ")";
173
174// ## Pre-release Version
175// Hyphen, followed by one or more dot-separated pre-release version
176// identifiers.
177inline const std::string PRE_RELEASE = "(?:-(" + PRE_RELEASE_IDENTIFIER
178 + R"((?:\.)" + PRE_RELEASE_IDENTIFIER
179 + ")*))";
180
181// ## Build Metadata Identifier
182// Any combination of digits, letters, or hyphens.
183inline const std::string BUILD_IDENTIFIER = "[0-9A-Za-z-]+";
184
185// ## Build Metadata
186// Plus sign, followed by one or more period-separated build metadata
187// identifiers.
188inline const std::string BUILD =
189 R"((?:\+()" + BUILD_IDENTIFIER + R"((?:\.)" + BUILD_IDENTIFIER + ")*))";
190
191// Note that the only major, minor, patch, and pre-release sections of
192// the version string are capturing groups. The build metadata is not a
193// capturing group, because it should not ever be used in version
194// comparison.
195inline const std::string FULL_PLAIN =
196 // "v?" +
197 MAIN_VERSION + PRE_RELEASE + "?" + BUILD + "?";
198
199// ## Interval of Version String
200// Something like ">1.2.0 and <=2.0.0".
201// A simple gt/lt/eq thing, or just "" to indicate "any version"
202inline const std::string GT_LT = "((?:<|>)?=?)";
203inline const std::string CLOSED_UNBOUNDED_INTERVAL =
204 "(^" + GT_LT + "?" + FULL_PLAIN + "$)";
205inline const std::string BOUNDED_INTERVAL =
206 "(^" + GT_LT + FULL_PLAIN + "( and )" + GT_LT + FULL_PLAIN + "$)";

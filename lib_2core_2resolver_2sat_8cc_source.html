6Vec<i32>
7to_assignments(const Vec<i32>& literals) {
8 Vec<i32> assignments;
9 for (auto&& l : literals | boost::adaptors::indexed()) {
10 const i32 literal = l.index() + 1;
11 if (l.value() != -1) {
12 assignments.emplace_back((l.value() % 2 == 0 ? 1 : -1) * literal);
13 } else { // for literals that can take either value, arbitrarily assign them
14 // to be true
15 assignments.emplace_back(literal);
16 }
17 }
18 return assignments;
19}

21// Find `1` or `-1` from whole clauses. Variables that have already been
22// assigned have been deleted from the clauses by the `delete_applied_literal`
23// function, so the index of the variable with the highest number of variables
24// is returned from the variables in the current clauses.
25i32
26maximum_literal_number_index(const Vec<Vec<i32>>& clauses) {
27 Map<i32, i32> frequency;
28 for (const auto& clause : clauses) {
29 for (const auto& literal : clause) {
30 auto result = frequency.insert({literal_to_index(literal), 1});
31 if (!result.second) {
32 result.first->second++;
33 }
34 }
35 }
36 auto x = std::max_element(
37 frequency.begin(), frequency.end(),
38 [](const auto& p1, const auto& p2) { return p1.second > p2.second; }
39 );
40 return x->first;
41}

43// Delete variables from the clauses for which variable assignment has been
44// determined.
45Status
46delete_set_literal(
47 Vec<Vec<i32>>& clauses, const i32& index, const i32& set_val
48) {
49 for (auto itr1 = clauses.begin(); itr1 != clauses.end(); ++itr1) {
50 for (auto itr2 = itr1->begin(); itr2 != itr1->end(); ++itr2) {
51 // set_val -> unassigned(-1) -> always false
52 // set_val -> true(0) -> value == index + 1
53 // set_val -> false(1) -> value == -(index + 1)
54 if (set_val >= 0
55 && (set_val == 0 ? index + 1 : -1 * (index + 1)) == *itr2) {
56 clauses.erase(itr1);
57 --itr1; // reset iterator
58 if (clauses.empty()) {
59 return Status::satisfied;
60 }
61 break; // to the next clause
62 } else if (index == literal_to_index(*itr2)) { // the literal with
63 // opposite polarity
64 itr1->erase(itr2); // remove the literal from the clause
65 --itr2; // reset iterator
66 if (itr1->empty()) {
67 // unsatisfiable currently
68 return Status::unsatisfied;
69 }
70 break; // to the next clause
71 }
72 }
73 }
74 return Status::normal;
75}

77// unit resolution
78Status
79unit_propagate(Vec<Vec<i32>>& clauses, Vec<i32>& literals) {
80 bool unit_clause_found = true;
81 while (unit_clause_found) {
82 unit_clause_found = false;
83
84 for (auto itr = clauses.begin(); itr != clauses.end(); ++itr) {
85 if (itr->size() == 1) { // unit clause ({3}, {5}, ...)
86 unit_clause_found = true;
87
88 // 0 - if true, 1 - if false, set the literal
89 literals[literal_to_index(*itr->begin())] = *itr->begin() < 0;
90
91 const i32 index = literal_to_index(*itr->begin());
92 Status result = delete_set_literal(clauses, index, literals[index]);
93 if (result == Status::satisfied || result == Status::unsatisfied) {
94 return result;
95 }
96 // Since another unit clause may have been created by this deletion,
97 // loop again from the beginning.
98 break;
99 } else if (itr->empty()) {
100 // the formula is unsatisfiable in this branch
101 return Status::unsatisfied;
102 }
103 }
104 }
105 return Status::normal;
106}

108// recursive DPLL algorithm
109[[nodiscard]] Result<Vec<i32>, String>
110dpll(Vec<Vec<i32>>& clauses, Vec<i32>& literals) {
111 if (clauses.empty()) {
112 return Ok(to_assignments(literals));
113 } else if (Status result = unit_propagate(clauses, literals);
114 result == Status::satisfied) {
115 return Ok(to_assignments(literals));
116 } else if (result == Status::unsatisfied) {
117 return Err(
118 "could not solve dependencies.\n"
119 "detail: given SAT problem was unsatisfied."
120 );
121 }
122
123 // The variable with the greatest frequency will be the next value assigned.
124 const i32 i = maximum_literal_number_index(clauses);
125 // need to apply twice, once true, the other false
126 for (i32 j = 0; j < 2; ++j) {
127 // copy the formula before recursive circulation
128 Vec<i32> new_literals = literals;
129
130 // if the number of literals with positive polarity are greater
131 // cond ? positive : negative
132 new_literals[i] =
133 calc_literal_polarity(clauses, i + 1) > 0 ? j : (j + 1) % 2;
134
135 // apply the change to all the clauses
136 if (Status result = delete_set_literal(clauses, i, new_literals[i]);
137 result == Status::satisfied) {
138 return Ok(to_assignments(new_literals));
139 } else if (result == Status::unsatisfied) {
140 // in this branch, return normally
141 continue;
142 }
143
144 const auto result = dpll(clauses, new_literals);
145 if (result.is_ok()) {
146 return result;
147 }
148 }
149 return Err(
150 "could not solve dependencies.\n"
151 "detail: given SAT problem was unsatisfied."
152 );
153}

80Lexer::component() {
81 // e.g. abcde
82 if (is_alphabet(this->one())) {
83 const size_type start = this->c1_index;
84 while (is_alpha_numeric(this->one())) {
85 this->step();
86 }
87 std::string_view sub = str.substr(start, this->c1_index - start);
88 return Token{Token::AlphaNumeric, sub};
89 }
90
91 // exactly zero
92 if (const auto [c1, c2] = this->two(); c1 == '0' && !is_digit(c2)) {
93 this->step();
94 return Token{Token::Numeric, 0};
95 }
96
97 const size_type start = this->c1_index;
98 while (is_digit(this->one())) {
99 this->step();
100 }
101 if (str[start] != '0' && !is_alphabet(this->one())) {
102 // e.g. 3425

104 const std::uint_fast64_t value = str_to_uint(sub).value();

106 }

108 // e.g. 3425dec85

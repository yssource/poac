205constexpr Token::Kind
206to_kind(StringRef kind) {
207 if (kind == "(") {
208 return Token::LeftParen;
209 } else if (kind == ")") {
210 return Token::RightParen;
211 } else if (kind == ",") {
212 return Token::Comma;
213 } else if (kind == "=") {
214 return Token::Equals;
215 } else if (kind == ">") {
216 return Token::Gt;
217 } else if (kind == ">=") {
218 return Token::GtEq;
219 } else if (kind == "<") {
220 return Token::Lt;
221 } else if (kind == "<=") {
222 return Token::LtEq;
223 } else {
224 throw exception("Unexpected to_kind error");
225 }
226}
